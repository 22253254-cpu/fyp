import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
import seaborn as sns

# 设置全局图表风格，使其更具学术/商业报告质感
sns.set_theme(style="whitegrid")
plt.rcParams['font.sans-serif'] = ['Arial']

# 1. 加载数据 (请确保文件名与您本地一致)
hk_df = pd.read_csv('hk_skincare_data.csv')
ml_df = pd.read_csv('mainland_skincare_data.csv')

# --- 数据清洗与特征工程 ---

# 1.1 统一性别标签
def get_hk_gender(val):
    if pd.isna(val): return None
    if val == 'Male': return 'Male'
    if val == 'Female': return 'Female'
    return None

hk_df['Gender_Clean'] = hk_df['1. What is your gender?'].apply(get_hk_gender)
ml_df['Gender_Clean'] = ml_df['Gender'].map({1: 'Female', 2: 'Male'}) 

# 1.2 统一消费阶层 (划分为 <100, 100-500, >500)
def get_hk_spending(val):
    if 'Less than 100' in str(val): return '<100'
    elif 'more than 500' in str(val): return '>500'
    elif pd.isna(val): return None
    else: return '100-500'

hk_df['Spending_Tier'] = hk_df['3. Which best describes your current skincare spending per month?'].apply(get_hk_spending)

def get_ml_spending(val):
    if val == 1: return '<100'
    elif val == 4: return '>500'
    else: return '100-500'
    
ml_df['Spending_Tier'] = ml_df['MonthlySpending'].apply(get_ml_spending)

# 1.3 统一产品使用情况 (提取洗面奶、爽肤水、面霜、精华/眼霜)
hk_prod_col = '2. Which of the following skincare product categories do you regularly use? (Select all that apply)'
hk_df['Use_Cleanser'] = hk_df[hk_prod_col].fillna('').str.contains('Cleanser', case=False).astype(int)
hk_df['Use_Toner'] = hk_df[hk_prod_col].fillna('').str.contains('Toner', case=False).astype(int)
hk_df['Use_Moisturizer'] = hk_df[hk_prod_col].fillna('').str.contains('Moisturizer', case=False).astype(int)
hk_df['Use_Serum_Eye'] = (hk_df[hk_prod_col].fillna('').str.contains('Serum', case=False) | hk_df[hk_prod_col].fillna('').str.contains('Eye Cream', case=False)).astype(int)

ml_df['Use_Cleanser'] = ml_df['cleanser'].fillna(0).astype(int)
ml_df['Use_Toner'] = ml_df['Toner'].fillna(0).astype(int)
ml_df['Use_Moisturizer'] = ml_df['Moisturizer'].fillna(0).astype(int)
ml_df['Use_Serum_Eye'] = (ml_df['Serum'].fillna(0).astype(int) | ml_df['Eye Cream'].fillna(0).astype(int))

# 1.4 合并数据
df = pd.concat([
    hk_df[['Gender_Clean', 'Spending_Tier', 'Use_Cleanser', 'Use_Toner', 'Use_Moisturizer', 'Use_Serum_Eye']], 
    ml_df[['Gender_Clean', 'Spending_Tier', 'Use_Cleanser', 'Use_Toner', 'Use_Moisturizer', 'Use_Serum_Eye']]
]).dropna(subset=['Gender_Clean'])

# 2. 统一 Target: Gender (Female: 0, Male: 1)
def get_hk_gender(val):
    if pd.isna(val): return None
    if val == 'Male': return 1 
    if val == 'Female': return 0 
    return None
hk_df['Target_Gender'] = hk_df['1. What is your gender?'].apply(get_hk_gender)
ml_df['Target_Gender'] = ml_df['Gender'].map({1: 0, 2: 1}) 

# 3. 提取并统一 Features
hk_cols = {
    'Price/Value for Money': [c for c in hk_df.columns if 'Price/Value for Money' in c][0],
    'Brand Reputation': [c for c in hk_df.columns if 'Brand Reputation' in c][0],
    'Ingredients & Efficacy': [c for c in hk_df.columns if 'Specific Ingredients' in c][0],
    'Product Reviews': [c for c in hk_df.columns if 'Product Reviews' in c][0],
    'Eco-friendly Packaging': [c for c in hk_df.columns if 'Eco-friendly' in c][0],
    'Clean/Non-Toxic Formula': [c for c in hk_df.columns if 'Non-Toxic' in c][0],
    'Ethical Values': [c for c in hk_df.columns if 'Ethical Values' in c][0]
}

ml_cols_mapping = {
    'Price/Value for Money': 'Price/Value for Money',
    'Brand Reputation': 'Brand Reputation',
    'Ingredients & Efficacy': 'Specific Ingredients / Efficacy',
    'Product Reviews': 'Product Reviews & Recommendations',
    'Eco-friendly Packaging': 'Packaging Design (whether Eco-friendly / Sustainable)',
    'Clean/Non-Toxic Formula': 'Clean / Non-Toxic Formula',
    'Ethical Values': "Brand's Ethical Values"
}

# 组合并重命名
hk_features = hk_df[list(hk_cols.values())].rename(columns={v: k for k, v in hk_cols.items()})
hk_features['Target_Gender'] = hk_df['Target_Gender']

ml_features = ml_df[list(ml_cols_mapping.values())].rename(columns={v: k for k, v in ml_cols_mapping.items()})
ml_features['Target_Gender'] = ml_df['Target_Gender']

combined_df = pd.concat([hk_features, ml_features]).dropna(subset=['Target_Gender'])

# 清洗可能存在的非标准字符 (例如把 "2, 3" 这种填错的变成数字 2)
def clean_numeric(val):
    try:
        if pd.isna(val): return float('nan')
        if isinstance(val, str):
            val = val.split(',')[0].strip()
        return float(val)
    except:
        return float('nan')

for col in hk_cols.keys():
    combined_df[col] = combined_df[col].apply(clean_numeric)

# 4. 准备机器学习模型所需数据
X = combined_df.drop('Target_Gender', axis=1)
X = X.fillna(X.median())  # 使用中位数填充缺失的打分
y = combined_df['Target_Gender']

# 5. 训练随机森林分类器
rf_clf = RandomForestClassifier(n_estimators=100, random_state=42, class_weight='balanced')
rf_clf.fit(X, y)

# 6. 提取特征重要性并绘图
importances = rf_clf.feature_importances_
feat_importances = pd.Series(importances, index=X.columns).sort_values(ascending=True)

plt.figure(figsize=(10, 6))
# 使用醒目的颜色凸显最重要的因素
colors = ['#bdc3c7' if i < len(feat_importances)-2 else '#e74c3c' for i in range(len(feat_importances))]
feat_importances.plot(kind='barh', color=colors, edgecolor='black', linewidth=1.2)

plt.title('Pseudo-Conjoint Analysis: Attribute Utility Gap\n(Feature Importance in Predicting Gender)', 
          fontsize=14, fontweight='bold', pad=15)
plt.xlabel('Feature Importance Score (Relative Decision Weight)', fontsize=12)
plt.ylabel('Skincare Product Attributes', fontsize=12)

# 在柱子上添加具体数值
for i, v in enumerate(feat_importances):
    plt.text(v + 0.002, i, f"{v:.3f}", color='black', va='center', fontweight='bold')

plt.tight_layout()
plt.show()

# 3. 特征工程 (提取和统一我们需要投入聚类的维度)

# 3.1 护肤步骤数 (Skincare Steps)
hk_prod_col = '2. Which of the following skincare product categories do you regularly use? (Select all that apply)'
def count_hk_steps(val):
    if pd.isna(val): return 0
    # 计算逗号数量+1来估算使用了几个产品
    return len(str(val).split(','))
hk_male['Skincare_Steps'] = hk_male[hk_prod_col].apply(count_hk_steps)

ml_prod_cols = ['cleanser', 'Toner', 'Moisturizer', 'Serum', 'Face Oil', 'Sunscreen', 'Eye Cream', 'Exfoliator (AHA/BHA)']
ml_male['Skincare_Steps'] = ml_male[ml_prod_cols].fillna(0).sum(axis=1)

# 3.2 消费阶梯量化 (Spending Tier 1-4)
def quantify_hk_spending(val):
    if 'Less than 100' in str(val): return 1
    elif '100 - 200' in str(val): return 2
    elif '200 - 500' in str(val): return 3
    elif 'more than 500' in str(val): return 4
    return 2 # 默认中位数
hk_male['Spending_Score'] = hk_male['3. Which best describes your current skincare spending per month?'].apply(quantify_hk_spending)
ml_male['Spending_Score'] = ml_male['MonthlySpending'] # 内地数据刚好是 1,2,3,4

hk_cols = {

# 4. K-means 聚类
# 必须先标准化数据，因为 "步骤数(1-6)" 和 "打分(1-5)" 尺度不同
scaler = StandardScaler()
X_scaled = scaler.fit_transform(male_df)

# 设置 K=2 进行聚类
kmeans = KMeans(n_clusters=2, random_state=42, n_init=10)
male_df['Cluster'] = kmeans.fit_predict(X_scaled)

# 为了让标签更稳定，我们判断哪个 Cluster 的护肤步骤更多，将其命名为 "Refined Men"
cluster_means = male_df.groupby('Cluster').mean()
if cluster_means.loc[0, 'Skincare_Steps'] > cluster_means.loc[1, 'Skincare_Steps']:
    cluster_labels = {0: 'Cluster 1 (Refined Men)', 1: 'Cluster 0 (Minimalist Men)'}
else:
    cluster_labels = {1: 'Cluster 1 (Refined Men)', 0: 'Cluster 0 (Minimalist Men)'}
    
male_df['Persona'] = male_df['Cluster'].map(cluster_labels)

# 打印聚类结果的人数
print(male_df['Persona'].value_counts())
