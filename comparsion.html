# --- 清洗香港数据 (HK Data Cleaning) ---
# 定义函数: 清洗长表头 (提取 [ ] 中的核心词)
def clean_hk_header(col_name):
    c = str(col_name).lower()
    if 'important' in c and '[' in c:
        match = re.search(r'\[(.*?)\]', c)
        if match:
            content = match.group(1).lower()
            if 'price' in content: return 'Price'
            if 'brand' in content: return 'Brand Reputation'
            if 'efficacy' in content or 'ingredient' in content: return 'Efficacy'
            if 'review' in content: return 'Reviews'
            if 'packaging' in content: return 'Sustainability (Packaging)'
            if 'clean' in content: return 'Clean Formula'
            if 'ethical' in content: return 'Ethical Values'
            if 'origin' in content: return 'Origin'
    return None

# 定义函数: 清洗评分值 ("第 1 列" -> 1.0)
def clean_hk_value(val):
    if pd.isna(val): return np.nan
    s = str(val)
    if '第 1 列' in s: return 1.0 # 修复常见问卷导出错误
    nums = re.findall(r'\d+', s)
    if nums: return float(nums[0])
    return np.nan

# 执行HK清洗
hk_clean = {}
for col in df_hk.columns:
    new_name = clean_hk_header(col)
    if new_name:
        hk_clean[new_name] = df_hk[col].apply(clean_hk_value)

df_hk_final = pd.DataFrame(hk_clean)
df_hk_final['Region'] = 'Hong Kong'

# --- 清洗内地数据 (Mainland Data Cleaning) ---
# 建立映射字典: 将内地数据的列名映射为标准英文名
ml_mapping = {
    'Price/Value for Money': 'Price',
    'Brand Reputation': 'Brand Reputation',
    'Specific Ingredients / Efficacy': 'Efficacy',
    'Product Reviews & Recommendations': 'Reviews',
    'Packaging Design (whether Eco-friendly / Sustainable)': 'Sustainability (Packaging)',
    'Clean / Non-Toxic Formula': 'Clean Formula',
    "Brand's Ethical Values": 'Ethical Values',
    'Origin': 'Origin'
}

# 执行映射和重命名
df_ml_final = df_ml.rename(columns=ml_mapping)
# 只保留需要的列
common_cols = list(ml_mapping.values())
df_ml_final = df_ml_final[common_cols].copy()
df_ml_final['Region'] = 'Mainland GBA'

# --- 合并数据 (Merge) ---
df_merged = pd.concat([df_ml_final, df_hk_final], ignore_index=True)
# 确保所有评分列都是数字
for col in common_cols:
    df_merged[col] = pd.to_numeric(df_merged[col], errors='coerce')

print("Data merged and cleaned successfully.")

# 2. 显著性检验 (T-Test)
# ==========================================
print("\n--- Step 2: Statistical Significance (T-Test) ---")

ttest_results = []
for col in common_cols:
    # 提取两组数据
    group_hk = df_merged[df_merged['Region']=='Hong Kong'][col].dropna()
    group_ml = df_merged[df_merged['Region']=='Mainland GBA'][col].dropna()
    
    # 执行独立样本T检验
    t_stat, p_val = stats.ttest_ind(group_hk, group_ml, equal_var=False)
    
    # 标记显著性水平
    sig = '***' if p_val < 0.001 else '**' if p_val < 0.01 else '*' if p_val < 0.05 else 'ns'
    
    ttest_results.append({
        'Factor': col,
        'HK Mean': round(group_hk.mean(), 2),
        'Mainland Mean': round(group_ml.mean(), 2),
        'P-Value': round(p_val, 4),
        'Significance': sig
    })

df_res = pd.DataFrame(ttest_results).sort_values('P-Value')
print(df_res)
print("\nNote: '*' indicates p < 0.05 (Significant Difference)")

print("\n--- Step 5: Clustering Analysis ---")

# 准备数据 (填充缺失值)
X = df_merged[common_cols].fillna(df_merged[common_cols].mean())
scaler = StandardScaler()

# K-Means 聚类 (3类)
kmeans = KMeans(n_clusters=3, random_state=42, n_init=10)
df_merged['Cluster'] = kmeans.fit_predict(X_scaled)

# 绘制: 两地人群构成对比 (堆叠柱状图)
comp = pd.crosstab(df_merged['Cluster'], df_merged['Region'], normalize='index') * 100
ax = comp.plot(kind='bar', stacked=True, color=['#1f77b4', '#d62728'], figsize=(10, 6))

plt.title('Consumer Segment Composition by Region', fontsize=14)
