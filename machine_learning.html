import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.ensemble import RandomForestRegressor
from sklearn.tree import DecisionTreeClassifier, plot_tree
from sklearn.preprocessing import LabelEncoder
from sklearn.cluster import KMeans
import re
import os

# ==========================================
# 0. 配置与样式
# ==========================================
plt.style.use('seaborn-v0_8-whitegrid')
plt.rcParams['font.family'] = 'sans-serif'
plt.rcParams['font.sans-serif'] = ['SimHei', 'Arial Unicode MS', 'DejaVu Sans']
plt.rcParams['axes.unicode_minus'] = False

# ==========================================
# 1. 数据加载 (Data Loading)
# ==========================================
print("--- Step 1: Loading Data ---")

def load_file(filename):
    if os.path.exists(filename):
        try:
            return pd.read_csv(filename) if filename.endswith('.csv') else pd.read_excel(filename)
        except: return None
    return None

df_ml = load_file('mainland_skincare_data.csv')
if df_ml is None: df_ml = load_file('mainland_skincare_data.xlsx - Sheet1.csv') 

df_hk = load_file('hk_skincare_data.csv')

if df_ml is None or df_hk is None:
    print("❌ 错误：找不到数据文件。")
else:
    print(f"✅ Data Loaded. Mainland: {df_ml.shape[0]} rows, HK: {df_hk.shape[0]} rows")

# ==========================================
# 2. 数据清洗与智能填充 (Cleaning & Imputation)
# ==========================================
print("\n--- Step 2: Cleaning & Imputing Missing Data ---")

# --- 清洗香港数据 ---
hk_data = {}
for col in df_hk.columns:
    c = str(col).lower()
    # 提取评分
    if 'important' in c and '[' in c:
        factor = re.search(r'\[(.*?)\]', c).group(1).lower()
        key = None
        if 'price' in factor: key = 'Price'
        elif 'brand' in factor: key = 'Brand'
        elif 'packaging' in factor: key = 'Packaging'
        elif 'clean' in factor: key = 'Clean'
        elif 'ethical' in factor: key = 'Ethical'
        
        if key: 
            hk_data[key] = df_hk[col].apply(lambda x: float(re.findall(r'\d+', str(x))[0]) if re.findall(r'\d+', str(x)) else np.nan)

    # 提取人口特征
    if 'old are you' in c: hk_data['Age'] = df_hk[col]
    elif 'gender' in c: hk_data['Gender'] = df_hk[col]
    elif 'spending' in c: hk_data['Spending'] = df_hk[col]
    elif 'profession' in c: hk_data['Profession'] = df_hk[col]

df_hk_ml = pd.DataFrame(hk_data)
df_hk_ml['Region'] = 'Hong Kong'

# --- 清洗内地数据 ---
ml_data = pd.DataFrame()
ml_mapping = {
    'Price/Value for Money': 'Price', 'Brand Reputation': 'Brand',
    'Packaging Design (whether Eco-friendly / Sustainable)': 'Packaging',
    'Clean / Non-Toxic Formula': 'Clean', "Brand's Ethical Values": 'Ethical',
    'age': 'Age', 'Gender': 'Gender', 'MonthlySpending': 'Spending', 'Profession': 'Profession'
}

for old_col, new_col in ml_mapping.items():
    matches = [c for c in df_ml.columns if c.lower() == old_col.lower()]
    if matches:
        ml_data[new_col] = df_ml[matches[0]]

ml_data['Region'] = 'Mainland'

# 合并数据
df_full = pd.concat([df_hk_ml, ml_data], ignore_index=True)

print(f"原始合并后数据量: {df_full.shape[0]} 行")
print(f"缺失值情况:\n{df_full.isnull().sum()}")

# --- 【关键策略】智能填充缺失值 ---
# 1. 数值型评分 (Price, Brand, Packaging...) -> 用该地区的平均分填充
score_cols = ['Price', 'Brand', 'Packaging', 'Clean', 'Ethical']
for col in score_cols:
    if col in df_full.columns:
        # 按地区分组计算均值，然后填充
        df_full[col] = df_full[col].fillna(df_full.groupby('Region')[col].transform('mean'))
        # 如果还有空值（比如整个地区都没人填），就用全局均值
        df_full[col] = df_full[col].fillna(df_full[col].mean())

# 2. 类别型特征 (Age, Gender...) -> 用"Unknown"或众数填充
cat_cols = ['Age', 'Gender', 'Spending', 'Profession']
for col in cat_cols:
    if col in df_full.columns:
        # 转为字符串，空值填为 "Unknown"
        df_full[col] = df_full[col].astype(str).replace('nan', 'Unknown')

print("✅ 智能填充完成，数据已无缺失。")

# ==========================================
# 3. 特征工程 (Feature Engineering)
# ==========================================
# 目标变量
df_full['Sustainability_Score'] = df_full[['Packaging', 'Clean', 'Ethical']].mean(axis=1)

# 编码
encoders = {}
feature_cols = []
for col in ['Region'] + cat_cols:
    le = LabelEncoder()
    new_col = f'{col}_Code'
    df_full[new_col] = le.fit_transform(df_full[col])
    feature_cols.append(new_col)

# ==========================================
# 4. 机器学习模型 (ML Models)
# ==========================================
print("\n--- Step 3: Running ML Models ---")

X = df_full[feature_cols]
y = df_full['Sustainability_Score']

# 模型一: 随机森林
rf = RandomForestRegressor(n_estimators=100, random_state=42)
rf.fit(X, y)

# 绘图: 特征重要性
importances = rf.feature_importances_
indices = np.argsort(importances)[::-1]
feature_names_clean = [col.replace('_Code', '') for col in feature_cols]

plt.figure(figsize=(10, 6))
plt.title('Key Drivers of Sustainability Preference (After Imputation)', fontsize=14, weight='bold')
plt.bar(range(X.shape[1]), importances[indices], align='center', color='#2ca02c', alpha=0.8)
plt.xticks(range(X.shape[1]), np.array(feature_names_clean)[indices], rotation=45)
plt.ylabel('Importance Score')
plt.tight_layout()
plt.show()

# 模型二: 决策树 (用于分类)
# K-Means 聚类
kmeans = KMeans(n_clusters=3, random_state=42, n_init=10)
df_full['Cluster'] = kmeans.fit_predict(df_full[score_cols])

dt = DecisionTreeClassifier(max_depth=3, random_state=42)
dt.fit(df_full[score_cols], df_full['Cluster'])

plt.figure(figsize=(16, 8))
plot_tree(dt, feature_names=score_cols, class_names=['Cluster 0', 'Cluster 1', 'Cluster 2'], 
          filled=True, rounded=True, fontsize=10)
plt.title('Decision Tree Segmentation Rules', fontsize=16)
plt.show()

# 输出结论
top_driver = np.array(feature_names_clean)[indices][0]
print(f"\n===== 📊 Analysis Summary =====")
print(f"1. Total Samples Analyzed: {len(df_full)} (All missing data successfully handled)")
print(f"2. Top Driver: '{top_driver}' is the most significant factor predicting sustainability preference.")
