import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
import re

# ==========================================
# 0. 全局配置 (Setup)
# ==========================================
# 设置绘图风格，保持专业性
plt.style.use('seaborn-v0_8-whitegrid')
plt.rcParams['font.family'] = 'sans-serif'
# 设置字体以支持中文 (根据系统自动回退)
plt.rcParams['font.sans-serif'] = ['SimHei', 'Arial Unicode MS', 'DejaVu Sans']
plt.rcParams['axes.unicode_minus'] = False

# ==========================================
# 1. 数据加载与预处理 (Data Loading & Cleaning)
# ==========================================
print("--- Step 1: Loading & Cleaning HK Data ---")

file_path = 'hk_skincare_data.csv'  # 请确认文件名

try:
    df = pd.read_csv(file_path)
    print(f"原始数据加载成功: {df.shape[0]} 行")
except FileNotFoundError:
    print(f"错误: 找不到文件 {file_path}")
    df = pd.DataFrame()

# --- 1.1 列名清洗 (处理长问题) ---
hk_data = {}

# 提取人口统计与行为特征
# 使用模糊匹配，避免硬编码长字符串
for col in df.columns:
    c_str = str(col).lower()
    
    # 提取评分题 (Matrix Questions)
    if 'important' in c_str and '[' in c_str:
        # 提取括号内的核心词，如 [Price]
        match = re.search(r'\[(.*?)\]', str(col))
        if match:
            factor = match.group(1).strip()
            # 简化列名
            if 'Price' in factor: key = 'Price'
            elif 'Brand' in factor: key = 'Brand'
            elif 'Ingredients' in factor or 'Efficacy' in factor: key = 'Efficacy'
            elif 'Review' in factor: key = 'Reviews'
            elif 'Packaging' in factor: key = 'Packaging'
            elif 'Clean' in factor: key = 'Clean_Formula'
            elif 'Ethical' in factor: key = 'Ethical'
            elif 'Origin' in factor: key = 'Origin'
            else: key = factor # 其他因素
            
            hk_data[key] = df[col]
            
    # 提取人口统计特征
    elif 'gender' in c_str: hk_data['Gender'] = df[col]
    elif 'old are you' in c_str: hk_data['Age'] = df[col]
    elif 'profession' in c_str: hk_data['Profession'] = df[col]
    elif 'skin type' in c_str: hk_data['SkinType'] = df[col]
    elif 'spending' in c_str: hk_data['Spending'] = df[col]
    elif 'how often' in c_str: hk_data['Frequency'] = df[col]

# 构建清洗后的 DataFrame
df_clean = pd.DataFrame(hk_data)

# --- 1.2 数值清洗与缺失值处理 (Handling Missing Values) ---
print("\n处理缺失值中...")

# 定义数值提取函数 (处理 "第 1 列" 或 "5 - Important")
def parse_rating(val):
    if pd.isna(val): return np.nan
    s = str(val)
    if '第 1 列' in s: return 1.0
    nums = re.findall(r'\d+', s)
    if nums: return float(nums[0])
    return np.nan

# 识别评分列 (数值型)
rating_cols = ['Price', 'Brand', 'Efficacy', 'Reviews', 'Packaging', 'Clean_Formula', 'Ethical', 'Origin']
rating_cols = [c for c in rating_cols if c in df_clean.columns]

# 1. 清洗数值格式
for col in rating_cols:
    df_clean[col] = df_clean[col].apply(parse_rating)

# 2. 填充数值型缺失值 -> 使用【均值填充】
# 逻辑：如果某人没填价格重要性，假设他处于平均水平
for col in rating_cols:
    mean_val = df_clean[col].mean()
    df_clean[col] = df_clean[col].fillna(mean_val)

# 3. 填充类别型缺失值 -> 使用【众数填充】或 "Unknown"
# 逻辑：没填性别不能填 0.5，填出现最多的性别
cat_cols = ['Gender', 'Age', 'Profession', 'SkinType', 'Spending', 'Frequency']
for col in cat_cols:
    if col in df_clean.columns:
        mode_val = df_clean[col].mode()[0]
        df_clean[col] = df_clean[col].fillna(mode_val).astype(str)

print("数据清洗完成！没有丢失任何数据。")
print(f"分析样本量: {len(df_clean)}")

# ==========================================
# 2. 复刻内地分析：基础画像 (Demographics)
# ==========================================
fig, axes = plt.subplots(2, 2, figsize=(16, 12))
fig.suptitle('Hong Kong Market: User Demographics', fontsize=18, weight='bold')

# 性别 (饼图)
if 'Gender' in df_clean.columns:
    counts = df_clean['Gender'].value_counts()
    axes[0,0].pie(counts, labels=counts.index, autopct='%1.1f%%', startangle=90, colors=sns.color_palette('pastel'))
    axes[0,0].set_title('Gender Distribution')

# 年龄 (条形图)
if 'Age' in df_clean.columns:
    sns.countplot(y='Age', data=df_clean, ax=axes[0,1], palette='viridis', order=sorted(df_clean['Age'].unique()))
    axes[0,1].set_title('Age Distribution')

# 职业 (条形图)
if 'Profession' in df_clean.columns:
    top_jobs = df_clean['Profession'].value_counts().head(5)
    sns.barplot(x=top_jobs.values, y=top_jobs.index, ax=axes[1,0], palette='mako')
    axes[1,0].set_title('Top 5 Professions')

# 肤质 (条形图)
if 'SkinType' in df_clean.columns:
    sns.countplot(x='SkinType', data=df_clean, ax=axes[1,1], palette='rocket')
    axes[1,1].set_title('Skin Type')

plt.tight_layout(rect=[0, 0.03, 1, 0.95])
plt.show()

# ==========================================
# 3. 复刻内地分析：消费行为 (Habits)
# ==========================================
fig, axes = plt.subplots(1, 2, figsize=(16, 6))

# 使用频率
if 'Frequency' in df_clean.columns:
    sns.countplot(y='Frequency', data=df_clean, ax=axes[0], palette='Blues_r')
    axes[0].set_title('Usage Frequency')

# 月消费
if 'Spending' in df_clean.columns:
    # 尝试按金额排序
    order = sorted(df_clean['Spending'].unique())
    sns.countplot(y='Spending', data=df_clean, ax=axes[1], palette='Greens_r', order=order)
    axes[1].set_title('Monthly Spending')

plt.tight_layout()
plt.show()

# ==========================================
# 4. 复刻内地分析：决策因素排名 (Preferences)
# ==========================================
if rating_cols:
    means = df_clean[rating_cols].mean().sort_values(ascending=True)
    
    plt.figure(figsize=(10, 6))
    bars = plt.barh(means.index, means.values, color='teal', alpha=0.8)
    plt.title('Importance of Purchase Decision Factors (Avg Score 1-5)', fontsize=16)
    plt.xlabel('Average Score')
    plt.xlim(0, 5.5)
    
    # 标注数值
    for bar in bars:
        plt.text(bar.get_width() + 0.05, bar.get_y() + bar.get_height()/2, 
                 f'{bar.get_width():.2f}', va='center')
    plt.tight_layout()
    plt.show()

# ==========================================
# 5. 复刻内地分析：K-Means 聚类 (Clustering)
# ==========================================
print("\n--- 执行 K-Means 聚类分析 ---")
# 这是一个非常重要的高级分析，能够识别细分市场

if rating_cols:
    # 1. 数据标准化 (Standardization)
    scaler = StandardScaler()
    X_scaled = scaler.fit_transform(df_clean[rating_cols])
    
    # 2. 聚类 (3类)
    kmeans = KMeans(n_clusters=3, random_state=42, n_init=10)
    clusters = kmeans.fit_predict(X_scaled)
    df_clean['Cluster'] = clusters
    
    # 3. 热力图展示各群组特征
    cluster_profile = df_clean.groupby('Cluster')[rating_cols].mean()
    
    plt.figure(figsize=(12, 5))
    sns.heatmap(cluster_profile, annot=True, cmap='YlGnBu', fmt=".2f", linewidths=.5)
    plt.title('Hong Kong Consumer Segments (Cluster Heatmap)', fontsize=16)
    plt.ylabel('Cluster ID')
    plt.show()
    
    print("聚类分析完成！请参考热力图识别不同人群的特征（例如：某一组可能对'Price'特别敏感）。")

import seaborn as sns

plt.rcParams['axes.unicode_minus'] = False

print("--- 正在处理数据 ---")
df_raw = pd.read_csv('hk_skincare_data.csv')

def clean_hk_header(col_name):
    if 'important' in str(col_name).lower() and '[' in str(col_name):
        match = re.search(r'\[(.*?)\]', str(col_name))
        if match: return match.group(1).strip()

    matches = df.filter(like=keyword)

        return matches.iloc[:, 0] # 只取第一列
    return pd.Series(index=df.index, dtype='object') # 返回空

data['Info_Raw'] = info_col.iloc[:,0] if not info_col.empty else ""

buy_col = df_raw.filter(like='Channels')

pref_cols = []

        elif 'Ingredients' in short: k='Specific Ingredients / Efficacy'
        elif 'Review' in short: k='Product Reviews & Recommendations'
        elif 'Packaging' in short: k='Packaging Design (Sustainable)'
        elif 'Clean' in short: k='Clean / Non-Toxic Formula'
